class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        
        unordered_map<int, int>ump;

        for(int i=0; i<nums.size(); i++ ){
            ump[ nums[i] ] = i;
        }

        int temp;
        for(int i=0; i<nums.size(); i++){
            temp = target - nums[i];

            if( ump.count(temp) && ump[temp] != i ){
                return {i, ump[temp]};
            }
        }

        return {};

    }
};
